git commands
git config --list [show git user details]
git config --global user.name "username" [to create git username]
git config --global user.email "email" [to create git email]
git init [to initialize git repository in pwd]
git status [list untracked files]
git add -A [adding all untracked files to staging area]
git commit -m "message" [from staging area to local repository]
git branch [list all branchs]
